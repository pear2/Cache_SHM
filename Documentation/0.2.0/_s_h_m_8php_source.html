60  /**

63  * Establishes a separate persistent storage. Adapter is automatically
64  * chosen based on the available extensions.
65  *

67  *
68  * @return static|SHM A new instance of an SHM adapter (child of this
69  * class).

71  final public static function factory($persistentId)
72  {
73  foreach (self::$_adapters as $adapter) {
74  try {
75  return new $adapter($persistentId);
76  } catch (E $e) {
77  //In case of a runtime error, try to fallback to other adapters.
78  }

82  1

96  /**
97  * Registers an adapter.
98  *
99  * Registers an SHM adapter, allowing you to call it with {@link factory()}.
100  *
101  * @param string $adapter FQCN of adapter. A valid adapter is one that
102  * extends this class. The class will be autoloaded if not already
103  * present.
104  * @param bool $prepend Whether to prepend this adapter into the list of
105  * possible adapters, instead of appending to it.
106  *
107  * @return bool TRUE on success, FALSE on failure.
108  */
109  final public static function registerAdapter($adapter, $prepend = false)
110  {
111  if (class_exists($adapter, true)
112  && is_subclass_of($adapter, '\\' . __CLASS__)
113  && $adapter::isMeetingRequirements()
114  ) {
115  if ($prepend) {
116  self::$_adapters = array_merge(
117  array($adapter),
118  self::$_adapters
119  );
120  } else {
121  self::$_adapters[] = $adapter;
122  }
123  return true;
124  }
125  return false;
126  }
